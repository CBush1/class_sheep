import random

eyes = [':', ';']
nose = ['^', '-']
mouth = ['(', ')', 'D']
number = 5
out_string = ''
for iteration in range(number):
    out_string = out_string + random.choice(eyes)
    out_string = out_string + random.choice(nose)
    out_string = out_string + random.choice(mouth)
print(out_string)
